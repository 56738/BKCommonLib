package org.bukkit.craftbukkit;

class CraftArt {
    public static org.bukkit.Art NotchToBukkit((Object) net.minecraft.server.Paintings art);

    public static (Object) net.minecraft.server.Paintings BukkitToNotch(org.bukkit.Art art);

    public static (Object) net.minecraft.server.Paintings NotchFromInternalId(int internalId) {
#if version >= 1.13.1
        return (net.minecraft.server.Paintings) net.minecraft.core.IRegistry.MOTIVE.fromId(internalId);
#elseif version >= 1.13
        return (net.minecraft.server.Paintings) net.minecraft.server.Paintings.a.getId(internalId);
#else
        net.minecraft.server.Paintings[] values = net.minecraft.server.Paintings.values();
        if (internalId >= 0 && internalId < values.length) {
            return values[internalId];
        }
        return net.minecraft.server.Paintings.KEBAB;
#endif
    }

    public static int NotchToInternalId((Object) net.minecraft.server.Paintings art) {
#if version >= 1.13.1
        return net.minecraft.core.IRegistry.MOTIVE.a(art);
#elseif version >= 1.13
        return net.minecraft.server.Paintings.a.a(art);
#else
        return art.ordinal();
#endif
    }

    public static (Object) net.minecraft.server.Paintings NotchFromInternalName(String internalName) {
#if version >= 1.13.1
        net.minecraft.resources.MinecraftKey key = new net.minecraft.resources.MinecraftKey(internalName);
        return (net.minecraft.server.Paintings) net.minecraft.core.IRegistry.MOTIVE.get(key);
#elseif version >= 1.13
        net.minecraft.resources.MinecraftKey key = new net.minecraft.resources.MinecraftKey(internalName);
        return (net.minecraft.server.Paintings) net.minecraft.server.Paintings.a.get(key);
#else
        net.minecraft.server.Paintings[] values = net.minecraft.server.Paintings.values();
        for (int i = 0; i < values.length; i++) {
            if (values[i].B.equals(internalName)) {
                return values[i];
            }
        }
        return net.minecraft.server.Paintings.KEBAB;
#endif
    }

    public static String NotchToInternalName((Object) net.minecraft.server.Paintings art) {
#if version >= 1.13.1
        net.minecraft.resources.MinecraftKey key = (net.minecraft.resources.MinecraftKey) net.minecraft.core.IRegistry.MOTIVE.getKey(art);
        if (key == null) {
            return null;
        } else {
            return key.getKey();
        }
#elseif version >= 1.13
        net.minecraft.resources.MinecraftKey key = (net.minecraft.resources.MinecraftKey) net.minecraft.server.Paintings.a.b(art);
        if (key == null) {
            return null;
        } else {
            return key.getKey();
        }
#else
        return art.B;
#endif
    }
}
